                    <div class="sold-out-overlay">
                        <span class="sold-out-text">HABIS!</span>
                    </div>
                ` : '';

                htmlContent += `
                    <div class="card-custom">
                        ${soldOutOverlay}
                        <img src="${product.image}" 
                            alt="${product.name}" 
                            onerror="this.onerror=null;this.src='${imageErrorPath}';" >
                        <h4>${product.name}</h4>
                        <p class="description">${product.description}</p>
                        <p class="price">Rp${formatRupiah(product.price)}</p>
                        
                        <button ${clickAction} ${buttonDisabled}> 
                            ${buttonText}
                        </button>
                    </div>
                `;
            }

            productList.innerHTML = htmlContent;
        }

        // --- Fungsi Animasi Hero Cookie ---
        function startCookieAnimation() {
            // Hanya ambil produk yang tersedia untuk animasi
            const availableProducts = [];
            for (let i = 0; i < products.length; i++) {
                if (products[i].status === 'available') {
                    availableProducts.push(products[i]);
                }
            }
            
            const imageSources = [];
            for (let i = 0; i < availableProducts.length; i++) {
                imageSources.push(availableProducts[i].image);
            }
            
            if (imageSources.length === 0) return; // Hentikan jika tidak ada produk tersedia
            
            let currentIndex = 0;

            // Set interval untuk ganti gambar
            setInterval(() => {
                currentIndex = (currentIndex + 1) % imageSources.length;
                
                heroCookieImage.classList.remove('spinning');
                heroCookieImage.src = imageSources[currentIndex];

                setTimeout(() => {
                    heroCookieImage.classList.add('spinning');
                }, 50);

            }, 5000); // Ganti gambar setiap 5 detik
        }
        
        // --- Fungsi Form Handling Baru ---
        function handleFormSubmit(event) {
            event.preventDefault(); // Mencegah refresh halaman
            
            const form = event.target;
            const name = form.name.value;
            const email = form.email.value;
            const message = form.message.value;

            // Tampilkan pesan sukses sementara
            formFeedback.textContent = `Terima kasih, ${name}! Pesan Anda telah kami terima. Kami akan membalas ke ${email} secepatnya.`;
            formFeedback.style.display = 'block';
            
            // Reset formulir setelah beberapa detik
            setTimeout(() => {
                form.reset();
                formFeedback.style.display = 'none';
                showToast(`Pesan dari ${name} berhasil dikirim! Kami akan segera memproses. 🥳`);
            }, 2000); 
        }

        // --- Fungsi Utilitas ---
        function formatRupiah(number) {
            // Fungsi untuk format angka
            return new Intl.NumberFormat('id-ID', { maximumFractionDigits: 0 }).format(number);
        }

        // Fungsi Toast (Pengganti alert)
        function showToast(message, type = 'default') {
            const toastContainer = document.getElementById('toast-container');
            const toast = document.createElement('div');
            
            toast.className = `toast ${type}`;
            toast.innerHTML = `<p>${message}</p>`;

            toastContainer.appendChild(toast);

            requestAnimationFrame(() => {
                toast.classList.add('show');
            });

            setTimeout(() => {
                toast.classList.remove('show');
                setTimeout(() => {
                    toastContainer.removeChild(toast);
                }, 500); 
            }, 4000);
        }

        // --- Event Listeners Awal ---
        window.onload = function() {
            renderProducts();
            startCookieAnimation(); 

            // Listener untuk tombol tutup modal
            closeModalButton.addEventListener('click', closeProductDetailsModal);

            // Menutup modal jika klik di luar area konten
            detailModal.addEventListener('click', (e) => {
                if (e.target.id === 'product-detail-modal') {
                    closeProductDetailsModal();
                }
            });
        };
